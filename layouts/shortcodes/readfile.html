{{- $file := .Get "file" | default (.Get 0) -}} {{- $code := .Get "code" | default false -}} {{- $lang := .Get "lang" | default "" -}} {{- $content := readFile $file -}} {{- /* Strip YAML front matter (--- … ---) and HTML-comment-wrapped variant () */ -}} {{- $content = replaceRE `(?s)\A\s*\n` "" $content -}} {{- $content = replaceRE `(?s)\A---\n.*?---\n` "" $content -}} {{- if $code -}} {{ highlight $content $lang "" }} {{- else -}} {{ $content | markdownify }} {{- end -}}
